import sys

def solve():
    line = sys.stdin.readline()
    if not line:
        return
    n = int(line.strip())
    result = []
    while True:
        result.append(str(n))
        if n == 1:
            break
        if n % 2 == 0:
            n //= 2
        else:
            n = n * 3 + 1
    print(" ".join(result))

if __name__ == "__main__":
    solve()
