  self.name = name
        self.hp = hp
        self.attack = attack
        self.defense = defense

    def attack_target(self, target):
        damage = max(0, self.attack - target.defense)
        target.hp -= damage
        return damage

def start_battle():
    raccoon = Entity("Common Raccoon", 60, 12, 5)
    monster = Entity("Monster", 80, 10, 3)

    while raccoon.hp > 0 and monster.hp > 0:
        raccoon.attack_target(monster)
        if monster.hp <= 0:
            print("Common Raccoon Wins")
            break
            
        monster.attack_target(raccoon)
        if raccoon.hp <= 0:
            print("Monster Wins")
            break

if __name__ == "__main__":
    start_battle()
