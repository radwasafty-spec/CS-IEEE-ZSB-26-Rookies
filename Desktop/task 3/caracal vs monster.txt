   self.name = name
        self.hp = hp
        self.attack = attack
        self.defense = defense

    def attack_target(self, target):
        damage = max(0, self.attack - target.defense)
        target.hp -= damage
        return damage

def start_battle():
    caracal = Entity("Caracal", 55, 20, 4)
    monster = Entity("Monster", 80, 10, 3)

    while caracal.hp > 0 and monster.hp > 0:
        caracal.attack_target(monster)
        if monster.hp <= 0:
            print("Caracal Wins")
            break
            
        monster.attack_target(caracal)
        if caracal.hp <= 0:
            print("Monster Wins")
            break

if __name__ == "__main__":
    start_battle()
