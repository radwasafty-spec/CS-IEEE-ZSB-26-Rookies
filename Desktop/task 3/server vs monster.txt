self.name = name
        self.hp = hp
        self.attack = attack
        self.defense = defense

    def attack_target(self, target):
        damage = max(0, self.attack - target.defense)
        target.hp -= damage
        return damage

def start_battle():
    server = Entity("Server", 200, 5, 20)
    monster = Entity("Monster", 80, 10, 3)

    while server.hp > 0 and monster.hp > 0:
        server.attack_target(monster)
        if monster.hp <= 0:
            print("Server Wins")
            break
            
        monster.attack_target(server)
        if server.hp <= 0:
            print("Monster Wins")
            break

if __name__ == "__main__":
    start_battle()
